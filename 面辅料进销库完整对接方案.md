# 面辅料进销库完整对接方案

**创建时间**: 2026-01-31  
**模块**: 样衣开发 + BOM表 + 物料采购 + 库存管理  
**目标**: 实现完整的面辅料流转闭环

---

## 📊 系统架构图

```
样衣开发（BOM表创建）
    ↓ 【检查库存】
库存系统（MaterialStock）
    ↓ 有库存 → 提示"无需采购"
    ↓ 无库存 → 自动生成采购需求
物料采购（MaterialPurchase）
    ↓ 采购到货 → 入库
库存系统（增加库存）
    ↓ 库存预警 → 低于安全库存提示
```

```java
@Service
public class StyleBomServiceImpl extends ServiceImpl<StyleBomMapper, StyleBom> 
    implements StyleBomService {

    @Autowired
    private MaterialStockService materialStockService;

    @Override
    @Transactional
    public List<StyleBom> saveBomWithStockCheck(List<StyleBom> bomList, Integer productionQty) {
        if (bomList == null || bomList.isEmpty()) {
            return bomList;
        }

        for (StyleBom bom : bomList) {
            // 1. 计算需求量（考虑损耗）
            BigDecimal usagePerUnit = bom.getUsageAmount(); // 单件用量
            BigDecimal lossRate = bom.getLossRate() != null ? bom.getLossRate() : BigDecimal.ZERO;
            BigDecimal totalUsage = usagePerUnit
                .multiply(BigDecimal.valueOf(productionQty))
                .multiply(BigDecimal.ONE.add(lossRate.divide(BigDecimal.valueOf(100))));
            
            int requiredQty = totalUsage.intValue();

            // 2. 查询库存（精确匹配：物料编码+颜色+规格）
            MaterialStock stock = materialStockService.getOne(
                new LambdaQueryWrapper<MaterialStock>()
                    .eq(MaterialStock::getMaterialCode, bom.getMaterialCode())
                    .eq(StringUtils.hasText(bom.getColor()), 
                        MaterialStock::getColor, bom.getColor())
                    .eq(StringUtils.hasText(bom.getSize()), 
                        MaterialStock::getSize, bom.getSize())
                    .last("LIMIT 1")
            );

            // 3. 计算库存状态
            int availableStock = 0;
            int requiredPurchase = requiredQty;
            String stockStatus = "none";

            if (stock != null) {
                availableStock = stock.getQuantity() - stock.getLockedQuantity();
                if (availableStock >= requiredQty) {
                    stockStatus = "sufficient";    // 库存充足
                    requiredPurchase = 0;
                } else if (availableStock > 0) {
                    stockStatus = "insufficient";  // 库存不足
                    requiredPurchase = requiredQty - availableStock;
                } else {
                    stockStatus = "none";          // 无库存
                }
            }

            // 4. 更新BOM记录（扩展字段）
            bom.setStockStatus(stockStatus);
            bom.setAvailableStock(availableStock);
            bom.setRequiredPurchase(requiredPurchase);

            log.info("[BOM库存检查] 物料: {}, 需求: {}, 可用: {}, 需采购: {}, 状态: {}",
                bom.getMaterialName(), requiredQty, availableStock, requiredPurchase, stockStatus);
        }

        // 5. 批量保存
        this.saveBatch(bomList);
        return bomList;
    }

    @Override
    public Map<String, Object> getBomStockSummary(String styleId, Integer productionQty) {
        List<StyleBom> bomList = this.list(
            new LambdaQueryWrapper<StyleBom>().eq(StyleBom::getStyleId, styleId)
        );

        int totalItems = bomList.size();
        int sufficientCount = 0;
        int insufficientCount = 0;
        int noneCount = 0;
        int totalPurchaseRequired = 0;

        for (StyleBom bom : bomList) {
            // 重新计算库存状态（实时数据）
            // ... 同上逻辑
            switch (bom.getStockStatus()) {
                case "sufficient": sufficientCount++; break;
                case "insufficient": insufficientCount++; break;
                case "none": noneCount++; break;
            }
            totalPurchaseRequired += bom.getRequiredPurchase();
        }

        Map<String, Object> summary = new HashMap<>();
        summary.put("totalItems", totalItems);
        summary.put("sufficientCount", sufficientCount);
        summary.put("insufficientCount", insufficientCount);
        summary.put("noneCount", noneCount);
        summary.put("totalPurchaseRequired", totalPurchaseRequired);
        summary.put("needPurchase", noneCount > 0 || insufficientCount > 0);
        
        return summary;
    }
}
```

**MaterialPurchaseOrchestrator.java** - 新增编排器

```java
@Service
public class MaterialPurchaseOrchestrator {

    @Autowired
    private MaterialPurchaseService materialPurchaseService;

    @Autowired
    private MaterialStockService materialStockService;

    @Autowired
    private MaterialInboundService materialInboundService;

    /**
     * 采购到货确认 + 自动入库
     * @param purchaseId 采购单ID
     * @param arrivedQuantity 到货数量
     * @param warehouseLocation 仓库位置
     * @return 是否成功
     */
    @Transactional(rollbackFor = Exception.class)
    public boolean confirmArrivalAndInbound(
        String purchaseId, 
        Integer arrivedQuantity,
        String warehouseLocation
    ) {
        // 1. 查询采购单
        MaterialPurchase purchase = materialPurchaseService.getById(purchaseId);
        if (purchase == null) {
            log.error("[采购入库] 采购单不存在 - purchaseId: {}", purchaseId);
            return false;
        }

        // 2. 更新采购到货数量
        purchase.setArrivedQuantity(arrivedQuantity);
        purchase.setReceivedTime(LocalDateTime.now());
        materialPurchaseService.updateById(purchase);

        // 3. 生成入库记录
        MaterialInbound inbound = new MaterialInbound();
        inbound.setId(UUID.randomUUID().toString().replace("-", ""));
        inbound.setInboundNo("IB" + System.currentTimeMillis());
        inbound.setPurchaseId(purchaseId);
        inbound.setMaterialCode(purchase.getMaterialCode());
        inbound.setMaterialName(purchase.getMaterialName());
        inbound.setColor(purchase.getColor());
        inbound.setSize(purchase.getSize());
        inbound.setInboundQuantity(arrivedQuantity);
        inbound.setWarehouseLocation(warehouseLocation);
        inbound.setInboundTime(LocalDateTime.now());
        
        UserContext ctx = UserContext.get();
        if (ctx != null) {
            inbound.setOperatorId(ctx.getUserId());
            inbound.setOperatorName(ctx.getUsername());
        }

        materialInboundService.save(inbound);

        // 4. 更新库存（关键！）
        boolean stockUpdated = updateMaterialStock(
            purchase.getMaterialCode(),
            purchase.getColor(),
            purchase.getSize(),
            arrivedQuantity,
            warehouseLocation
        );

        if (stockUpdated) {
            // 5. 标记采购单已更新库存
            purchase.setStockUpdated(1);
            purchase.setStockUpdateTime(LocalDateTime.now());
            purchase.setInboundRecordId(inbound.getId());
            materialPurchaseService.updateById(purchase);
            
            log.info("[采购入库成功] 采购单: {}, 物料: {}, 入库数量: {}, 库存已更新",
                purchase.getPurchaseNo(), purchase.getMaterialName(), arrivedQuantity);
        } else {
            log.error("[采购入库失败] 库存更新失败 - purchaseId: {}", purchaseId);
            throw new RuntimeException("库存更新失败");
        }

        return true;
    }

    /**
     * 更新库存（增加库存）
     */
    private boolean updateMaterialStock(
        String materialCode,
        String color,
        String size,
        Integer increaseQty,
        String warehouseLocation
    ) {
        // 查询或创建库存记录
        MaterialStock stock = materialStockService.getOne(
            new LambdaQueryWrapper<MaterialStock>()
                .eq(MaterialStock::getMaterialCode, materialCode)
                .eq(StringUtils.hasText(color), MaterialStock::getColor, color)
                .eq(StringUtils.hasText(size), MaterialStock::getSize, size)
                .last("LIMIT 1")
        );

        if (stock == null) {
            // 首次入库，创建库存记录
            stock = new MaterialStock();
            stock.setId(UUID.randomUUID().toString().replace("-", ""));
            stock.setMaterialCode(materialCode);
            stock.setColor(color);
            stock.setSize(size);
            stock.setQuantity(increaseQty);
            stock.setLockedQuantity(0);
            stock.setSafetyStock(100); // 默认安全库存
            stock.setLocation(warehouseLocation);
            return materialStockService.save(stock);
        } else {
            // 库存增加
            stock.setQuantity(stock.getQuantity() + increaseQty);
            stock.setUpdateTime(LocalDateTime.now());
            return materialStockService.updateById(stock);
        }
    }
}
```

**MaterialStockAlertService.java**

```java
@Service
public class MaterialStockAlertService {

    @Autowired
    private MaterialStockService materialStockService;

    /**
     * 获取低库存预警列表
     * @return 低库存物料列表
     */
    public List<Map<String, Object>> getLowStockAlerts() {
        List<MaterialStock> lowStockList = materialStockService.list(
            new LambdaQueryWrapper<MaterialStock>()
                .apply("quantity - locked_quantity < safety_stock")
                .orderByAsc(MaterialStock::getQuantity)
        );

        List<Map<String, Object>> alerts = new ArrayList<>();
        for (MaterialStock stock : lowStockList) {
            int availableQty = stock.getQuantity() - stock.getLockedQuantity();
            int shortfall = stock.getSafetyStock() - availableQty;
            
            Map<String, Object> alert = new HashMap<>();
            alert.put("materialCode", stock.getMaterialCode());
            alert.put("materialName", stock.getMaterialName());
            alert.put("color", stock.getColor());
            alert.put("availableQty", availableQty);
            alert.put("safetyStock", stock.getSafetyStock());
            alert.put("shortfall", shortfall);
            alert.put("alertLevel", getAlertLevel(availableQty, stock.getSafetyStock()));
            alerts.add(alert);
        }

        return alerts;
    }

    /**
     * 获取预警等级
     */
    private String getAlertLevel(int availableQty, int safetyStock) {
        if (availableQty <= 0) {
            return "critical";  // 严重：无库存
        } else if (availableQty < safetyStock * 0.3) {
            return "high";      // 高：低于安全库存30%
        } else if (availableQty < safetyStock * 0.6) {
            return "medium";    // 中：低于安全库存60%
        } else {
            return "low";       // 低：低于安全库存
        }
    }

    /**
     * 自动生成采购建议
     */
    public List<Map<String, Object>> generatePurchaseSuggestions() {
        List<Map<String, Object>> alerts = getLowStockAlerts();
        
        List<Map<String, Object>> suggestions = new ArrayList<>();
        for (Map<String, Object> alert : alerts) {
            Map<String, Object> suggestion = new HashMap<>();
            suggestion.putAll(alert);
            
            // 建议采购量 = 安全库存 * 2 - 当前可用
            int availableQty = (int) alert.get("availableQty");
            int safetyStock = (int) alert.get("safetyStock");
            int suggestedQty = safetyStock * 2 - availableQty;
            
            suggestion.put("suggestedPurchaseQty", Math.max(suggestedQty, 0));
            suggestions.add(suggestion);
        }

        return suggestions;
    }
}
```

## 📊 监控指标

### 库存健康度
```
健康度 = (可用库存 / 安全库存) * 100%

90%+ : 优秀 🟢
60-90%: 良好 🟡
30-60%: 警告 🟠
<30% : 危险 🔴
```

### 采购效率
```
- 平均采购周期：采购创建 → 到货入库
- 库存周转率：出库量 / 平均库存
- 预警响应时间：触发预警 → 生成采购单
```
