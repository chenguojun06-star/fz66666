/**
 * 后端 0/1 → 前端 boolean
 */
export function intToBool(value: number | undefined | null): boolean | undefined {
  if (value === null || value === undefined) return undefined;
  return value === 1;
}

/**
 * 前端 boolean → 后端 0/1
 */
export function boolToInt(value: boolean | undefined): number | undefined {
  if (value === undefined) return undefined;
  return value ? 1 : 0;
}

// 使用示例
import { intToBool, boolToInt } from '@/services/common/typeConverter';

    const order = response.data;
    
    // 转换 0/1 → boolean
    return {
      ...order,
      progressWorkflowLocked: intToBool(order.progressWorkflowLocked),
      procurementManuallyCompleted: intToBool(order.procurementManuallyCompleted),
    };
  },
