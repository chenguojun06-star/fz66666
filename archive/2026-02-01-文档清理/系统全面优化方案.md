#### 2.2.2 数据结构优化

**优化点：**
- 使用Map替代List进行查找（O(n) → O(1)）
- 使用Set去重替代循环判断
- 使用Stream API优化集合操作

**示例：**
```java
// 优化前 - O(n^2)
List<Order> uniqueOrders = new ArrayList<>();
for (Order order : orders) {
    boolean exists = false;
    for (Order uo : uniqueOrders) {
        if (uo.getId().equals(order.getId())) {
            exists = true;
            break;
        }
    }
    if (!exists) uniqueOrders.add(order);
}

// 优化后 - O(n)
Map<String, Order> orderMap = orders.stream()
    .collect(Collectors.toMap(
        Order::getId,
        Function.identity(),
        (existing, replacement) -> existing
    ));
List<Order> uniqueOrders = new ArrayList<>(orderMap.values());
```

#### 4.1.2 页面加载优化

**请求队列管理：**

```javascript
// RequestQueue.js
class RequestQueue {
  constructor(maxConcurrent = 3) {
    this.maxConcurrent = maxConcurrent;
    this.queue = [];
    this.running = 0;
  }
  
  async add(requestFn, priority = 0) {
    if (this.running >= this.maxConcurrent) {
      await new Promise(resolve => {
        this.queue.push({ resolve, priority });
        this.queue.sort((a, b) => b.priority - a.priority);
      });
    }
    
    this.running++;
    try {
      return await requestFn();
    } finally {
      this.running--;
      if (this.queue.length > 0) {
        const next = this.queue.shift();
        next.resolve();
      }
    }
  }
}

// 使用示例
const queue = new RequestQueue(3);

// 高优先级请求
queue.add(() => api.production.getOrderDetail(id), 10);

// 低优先级请求
queue.add(() => api.production.getStatistics(), 1);
```

**数据缓存策略：**

```typescript
// CacheManager.ts
class CacheManager {
  private cache = new Map<string, CacheEntry>();
  private readonly DEFAULT_TTL = 5 * 60 * 1000; // 5分钟
  
  set(key: string, data: any, ttl = this.DEFAULT_TTL) {
    this.cache.set(key, {
      data,
      timestamp: Date.now(),
      ttl
    });
  }
  
  get(key: string) {
    const entry = this.cache.get(key);
    if (!entry) return null;
    
    if (Date.now() - entry.timestamp > entry.ttl) {
      this.cache.delete(key);
      return null;
    }
    
    return entry.data;
  }
  
  // 内存管理
  cleanup() {
    const now = Date.now();
    for (const [key, entry] of this.cache.entries()) {
      if (now - entry.timestamp > entry.ttl) {
        this.cache.delete(key);
      }
    }
  }
}
```

}) => {
  const [selectedIds, setSelectedIds] = useState<string[]>([]);
  
  // 全选/反选
  const toggleSelectAll = () => {
    if (selectedIds.length === items.length) {
      setSelectedIds([]);

      await onBatchUpdate(batch, { action });
    }
  };
  
  return (

      <Checkbox 
        checked={selectedIds.length === items.length}
        onChange={toggleSelectAll}
      >
        全选
      </Checkbox>

```typescript
// VirtualList.tsx - 增强版
interface VirtualListProps {
  items: any[];
  itemHeight: number | ((item: any) => number);
  renderItem: (item: any, index: number) => React.ReactNode;
  overscan?: number;
  onEndReached?: () => void;
}

const VirtualList: React.FC<VirtualListProps> = ({
  items,
  itemHeight,
  renderItem,
  overscan = 5,
  onEndReached
}) => {
  const containerRef = useRef<HTMLDivElement>(null);
  const [scrollTop, setScrollTop] = useState(0);
  const [containerHeight, setContainerHeight] = useState(0);
  
  // 动态高度计算
  const getItemHeight = useCallback((item: any, index: number) => {
    return typeof itemHeight === 'function' 
      ? itemHeight(item) 
      : itemHeight;
  }, [itemHeight]);
  
  // 计算可见范围
  const { virtualItems, totalHeight, offsetY } = useMemo(() => {
    let offset = 0;
    let startIndex = -1;
    let endIndex = -1;
    
    for (let i = 0; i < items.length; i++) {
      const height = getItemHeight(items[i], i);
      
      if (startIndex === -1 && offset + height > scrollTop) {
        startIndex = Math.max(0, i - overscan);
      }
      
      if (endIndex === -1 && offset > scrollTop + containerHeight) {
        endIndex = Math.min(items.length - 1, i + overscan);
        break;
      }
      
      offset += height;
    }
    
    if (endIndex === -1) endIndex = items.length - 1;
    
    return {
      virtualItems: items.slice(startIndex, endIndex + 1),
      totalHeight: offset,
      offsetY: startIndex > 0 
        ? items.slice(0, startIndex).reduce((sum, item, i) => 
            sum + getItemHeight(item, i), 0)
        : 0
    };
  }, [items, scrollTop, containerHeight, overscan, getItemHeight]);
  
  // 无限滚动
  useEffect(() => {
    if (virtualItems.length > 0 && 
        virtualItems[virtualItems.length - 1] === items[items.length - 1]) {
      onEndReached?.();
    }
  }, [virtualItems, items, onEndReached]);
  
  return (
    <div 
      ref={containerRef}
      className="virtual-list"
      onScroll={(e) => setScrollTop(e.currentTarget.scrollTop)}
    >
      <div style={{ height: totalHeight }}>
        <div style={{ transform: `translateY(${offsetY}px)` }}>
          {virtualItems.map((item, index) => (
            <div key={item.id} style={{ height: getItemHeight(item, index) }}>
              {renderItem(item, index)}
            </div>
          ))}
        </div>
      </div>
    </div>
  );
};
```

  }

// 插件管理器
class PluginManager {
  private plugins = new Map<string, Plugin>();
  private context: PluginContext;
  
  constructor(context: PluginContext) {
    this.context = context;
  }
  
  async loadPlugin(plugin: Plugin) {
    // 检查依赖
    if (plugin.dependencies) {
      for (const dep of plugin.dependencies) {
        if (!this.plugins.has(dep)) {
          throw new Error(`Plugin ${plugin.name} requires ${dep}`);
        }
      }
    }
    
    // 安装插件
    plugin.install(this.context);
    this.plugins.set(plugin.name, plugin);
    
    console.log(`Plugin ${plugin.name} v${plugin.version} loaded`);
  }
  
  async unloadPlugin(name: string) {
    const plugin = this.plugins.get(name);
    if (plugin && plugin.uninstall) {
      plugin.uninstall(this.context);
    }
    this.plugins.delete(name);
  }
}

        }
