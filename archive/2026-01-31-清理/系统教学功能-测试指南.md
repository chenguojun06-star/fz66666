```sql
USE fashion_supplychain;

-- 添加系统教学权限到字典表
INSERT INTO t_dict (dict_type, dict_label, dict_value, sort_order, status, remark, created_at, updated_at)
VALUES 
('permission_code', '系统教学', 'MENU_TUTORIAL', 1500, 1, '系统教学中心菜单权限', NOW(), NOW())
ON DUPLICATE KEY UPDATE updated_at = NOW();

-- 为管理员角色添加权限(假设管理员角色名为'管理员')
-- 方法1: 如果使用t_role_permission表
INSERT INTO t_role_permission (role_id, permission_code, created_at)
SELECT 
    r.id as role_id,
    'MENU_TUTORIAL' as permission_code,
    NOW() as created_at
FROM t_role r
WHERE r.role_name = '管理员'
ON DUPLICATE KEY UPDATE updated_at = NOW();

-- 方法2: 如果角色权限存储在t_role.permission_codes字段(JSON数组)
-- 需要手动在角色管理页面添加权限,或执行以下SQL
UPDATE t_role 
SET permission_codes = JSON_ARRAY_APPEND(permission_codes, '$', 'MENU_TUTORIAL')
WHERE role_name = '管理员' 
  AND NOT JSON_CONTAINS(permission_codes, '"MENU_TUTORIAL"');
```
