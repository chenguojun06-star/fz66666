```java
public Result<?> list(@RequestParam Map<String, Object> params) {
    // 1. 检查特殊参数，路由到专门方法
```

```typescript
// 自动重定向旧端点到新端点
export function adaptLegacyAPI(url: string, params: any) {
  // 示例：/by-order-no/PO123 → /list?orderNo=PO123
  if (url.match(/\/by-order-no\//)) {
    const orderNo = url.split('/').pop();
    return { url: '/list', params: { ...params, orderNo } };
  }
  return { url, params };
}
```
