## 1. 项目概述

### 1.1 项目背景

服装供应链管理系统是一个三端协同的企业级应用，覆盖 PC 管理后台、微信小程序工厂扫码、完整的业务编排后端。系统已具备 60+ 智能编排器和 65+ AI 工具，但智能化能力分散，尚未形成统一中枢。

### 1.2 升级目标

将系统从"静态业务系统"升级为"活系统"，具备五大核心能力：

| 能力 | 当前状态 | 目标状态 |
|------|---------|---------|
| **感知** | 部分埋点 | 全链路实时感知 |
| **判断** | 规则判断 | AI 态势融合 |
| **行动** | 仅显示建议 | 自动执行 + 闭环追踪 |
| **记忆** | 分散存储 | 统一知识库 + 向量召回 |
| **学习** | 无闭环 | 反馈学习 + 策略优化 |

### 1.3 升级范围

- **前端**：React 18 → Next.js 15 + React 19
- **后端**：Spring Boot 2.7 → 3.2 + Virtual Threads
- **数据库**：MySQL → PostgreSQL + pgvector
- **智能化**：五层智能架构全面升级

## 5. 详细设计

### 5.1 数据库设计

#### 5.1.1 智能事件表

```sql
CREATE TABLE t_intelligence_signal (
    id BIGSERIAL PRIMARY KEY,
    tenant_id BIGINT NOT NULL,
    signal_type VARCHAR(50) NOT NULL,        -- 事件类型
    biz_type VARCHAR(50),                     -- 业务类型
    biz_id BIGINT,                            -- 业务ID
    signal_level VARCHAR(20),                 -- 事件级别：INFO/WARNING/ERROR/CRITICAL
    signal_payload JSONB,                     -- 事件详情
    signal_time TIMESTAMPTZ DEFAULT NOW(),   -- 事件时间
    source_terminal VARCHAR(20),              -- 来源终端：PC/MINI_PROGRAM/API
    processed BOOLEAN DEFAULT FALSE,          -- 是否已处理
    processed_at TIMESTAMPTZ,                 -- 处理时间
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 索引
CREATE INDEX idx_signal_tenant_type ON t_intelligence_signal(tenant_id, signal_type);
CREATE INDEX idx_signal_time ON t_intelligence_signal(signal_time);
CREATE INDEX idx_signal_biz ON t_intelligence_signal(biz_type, biz_id);

-- 注释
COMMENT ON TABLE t_intelligence_signal IS '智能事件表 - 感知层核心表';
COMMENT ON COLUMN t_intelligence_signal.signal_type IS '事件类型：ORDER_CHANGE/SCAN_ANOMALY/STOCK_ALERT/FINANCE_DIFF/SYSTEM_ALERT';
```

#### 5.1.2 智能动作表

```sql
CREATE TABLE t_intelligence_action (
    id BIGSERIAL PRIMARY KEY,
    tenant_id BIGINT NOT NULL,
    signal_id BIGINT REFERENCES t_intelligence_signal(id),
    action_type VARCHAR(50) NOT NULL,         -- 动作类型
    action_level VARCHAR(10) DEFAULT 'L1',    -- 动作级别：L1提示/L2建议/L3软拦截/L4自动
    target_role VARCHAR(50),                  -- 目标角色
    target_user_id VARCHAR(64),               -- 目标用户
    action_status VARCHAR(20) DEFAULT 'PENDING', -- 状态：PENDING/EXECUTING/COMPLETED/FAILED/CANCELLED
    action_payload JSONB,                     -- 动作详情
    due_time TIMESTAMPTZ,                     -- 截止时间
    executed_at TIMESTAMPTZ,                  -- 执行时间
    executed_by VARCHAR(64),                  -- 执行人
    result_payload JSONB,                     -- 执行结果
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);
```

```java
/**
 * 订单健康度评分编排器 - 判断层核心

@Service
public class OrderHealthScoreOrchestrator {

    private static final double PROGRESS_WEIGHT = 0.40;
    private static final double DELIVERY_WEIGHT = 0.35;
    private static final double PROCUREMENT_WEIGHT = 0.25;

    @Autowired
    private ProductionOrderService orderService;
    
    @Autowired
    private IntelligenceMemoryOrchestrator memoryOrchestrator;

    /**
     * 计算订单健康度评分
     */
    public HealthScoreResult calculate(Long orderId) {
        ProductionOrder order = orderService.getById(orderId);
        
        // 1. 计算各维度分数
        double progressScore = calculateProgressScore(order);
        double deliveryScore = calculateDeliveryScore(order);
        double procurementScore = calculateProcurementScore(order);
        
        // 2. 加权计算总分
        double totalScore = progressScore * PROGRESS_WEIGHT 
                          + deliveryScore * DELIVERY_WEIGHT 
                          + procurementScore * PROCUREMENT_WEIGHT;
        
        // 3. 识别风险因素
        List<RiskFactor> riskFactors = identifyRiskFactors(order);
        
        // 4. 生成建议动作
        List<NextAction> nextActions = generateNextActions(order, riskFactors);
        
        // 5. 召回相似案例
        List<SimilarCase> similarCases = memoryOrchestrator.recallSimilarCases(order);
        
        return HealthScoreResult.builder()
            .orderId(orderId)
            .totalScore(totalScore)
            .progressScore(progressScore)
            .deliveryScore(deliveryScore)
            .procurementScore(procurementScore)
            .riskLevel(determineRiskLevel(totalScore))
            .riskFactors(riskFactors)
            .nextActions(nextActions)
            .similarCases(similarCases)
            .build();
    }

    /**
     * 计算进度分数
     */
    private double calculateProgressScore(ProductionOrder order) {
        Integer expectedProgress = order.getExpectedProgress();
        Integer actualProgress = order.getProductionProgress();
        
        if (expectedProgress == null || actualProgress == null) {
            return 100.0;
        }
        
        double deviation = (double) (expectedProgress - actualProgress) / expectedProgress;
        return Math.max(0, 100 - deviation * 100);
    }

    /**
     * 计算交付分数
     */
    private double calculateDeliveryScore(ProductionOrder order) {
        LocalDate deliveryDate = order.getDeliveryDate();
        LocalDate now = LocalDate.now();
        
        if (deliveryDate == null) {
            return 100.0;
        }
        
        long daysRemaining = ChronoUnit.DAYS.between(now, deliveryDate);
        Integer progress = order.getProductionProgress();
        
        if (progress == null) progress = 0;
        
        // 进度与剩余时间的匹配度
        double expectedProgressByTime = 100.0 * (1 - (double) daysRemaining / 30);
        double deviation = Math.abs(progress - expectedProgressByTime);
        
        return Math.max(0, 100 - deviation);
    }

    /**
     * 计算采购分数
     */
    private double calculateProcurementScore(ProductionOrder order) {
        // 检查物料状态
        List<MaterialStatus> materials = materialService.getByOrderId(order.getId());
        
        long missingCount = materials.stream()
            .filter(m -> "MISSING".equals(m.getStatus()))
            .count();
        
        long delayedCount = materials.stream()
            .filter(m -> "DELAYED".equals(m.getStatus()))
            .count();
        
        double penalty = missingCount * 20 + delayedCount * 10;
        return Math.max(0, 100 - penalty);
    }

    /**
     * 确定风险级别
     */
    private String determineRiskLevel(double score) {
        if (score >= 75) return "LOW";
        if (score >= 50) return "MEDIUM";
        if (score >= 25) return "HIGH";
        return "CRITICAL";
    }
}
```

```java
            case "L2" -> executeL2(action);  // 建议执行
            case "L3" -> executeL3(action);  // 软拦截
            case "L4" -> executeL4(action);  // 自动执行
        }
    }

    /**
     * L1: 仅提示
     */

        notificationService.sendAlert(action);
    }

    /**
     * L2: 建议执行
     */
    private void executeL2(IntelligenceAction action) {
        notificationService.sendAlert(action);
        // 在界面显示建议操作按钮
    }

    /**
     * L3: 软拦截（可手动继续）
     */
    private void executeL3(IntelligenceAction action) {
        notificationService.sendAlert(action);
        // 设置拦截标记，用户可手动解除
    }

    /**
     * L4: 自动执行
     */
    @Transactional(rollbackFor = Exception.class)
    private void executeL4(IntelligenceAction action) {
        try {
            action.setActionStatus("EXECUTING");
            actionService.updateById(action);
            
            // 执行具体动作
            ActionResult result = executeAction(action);
            
            action.setActionStatus("COMPLETED");
            action.setResultPayload(result.toJson());
            action.setExecutedAt(LocalDateTime.now());
            action.setExecutedBy("SYSTEM");
            
        } catch (Exception e) {
            action.setActionStatus("FAILED");
            action.setResultPayload(Map.of("error", e.getMessage()));
        }
        
        actionService.updateById(action);
    }

    /**
     * 执行具体动作
     */
    private ActionResult executeAction(IntelligenceAction action) {
        return switch (action.getActionType()) {
            case "REMIND" -> executeRemind(action);
            case "DISPATCH" -> executeDispatch(action);
            case "CREATE_PURCHASE" -> executeCreatePurchase(action);
            case "NOTIFY" -> executeNotify(action);
            default -> ActionResult.unknown("未知动作类型");
        };

            .type("REMIND")
            .orderId(orderId)
            .assignee(action.getTargetUserId())
            .dueTime(action.getDueTime())
            .build());
        
        // 3. 记录审计日志
        auditLogService.log("REMIND_SENT", orderId);
        
        return ActionResult.success("催办已发送");
    }
```

```java
        // 2. 更新建议统计

        // 3. 检查是否需要策略调优

            tuningService.triggerTuning(UserContext.tenantId());
        }
    }

    /**
     * 更新建议统计
     */
    private void updateSuggestionStats(Long suggestionId, boolean accepted) {
        IntelligenceSuggestion suggestion = suggestionService.getById(suggestionId);
        if (suggestion == null) return;
        
        // 更新采纳统计
        suggestion.setTotalFeedback(suggestion.getTotalFeedback() + 1);
        if (accepted) {
            suggestion.setAcceptedCount(suggestion.getAcceptedCount() + 1);
        }
        
        // 计算采纳率
        double adoptionRate = (double) suggestion.getAcceptedCount() / suggestion.getTotalFeedback();
        suggestion.setAdoptionRate(adoptionRate);
        
        suggestionService.updateById(suggestion);
    }

    /**
     * 评估建议效果
     */
    public SuggestionEffect evaluateEffect(Long suggestionId) {
        IntelligenceSuggestion suggestion = suggestionService.getById(suggestionId);
        List<IntelligenceFeedback> feedbacks = feedbackService.findBySuggestionId(suggestionId);
        
        return SuggestionEffect.builder()
            .suggestionId(suggestionId)
            .totalFeedbacks(feedbacks.size())
            .adoptionRate(calculateAdoptionRate(feedbacks))
            .accuracy(calculateAccuracy(feedbacks))
            .avgEffectivenessScore(calculateAvgEffectiveness(feedbacks))
            .impactScore(calculateImpact(feedbacks))
            .build();
    }

    /**
     * 计算采纳率
     */
    private double calculateAdoptionRate(List<IntelligenceFeedback> feedbacks) {
        if (feedbacks.isEmpty()) return 0.0;
        
        long accepted = feedbacks.stream()
            .filter(IntelligenceFeedback::getAccepted)
            .count();
        
        return (double) accepted / feedbacks.size();
    }

    /**
     * 计算准确率
     */
    private double calculateAccuracy(List<IntelligenceFeedback> feedbacks) {
        if (feedbacks.isEmpty()) return 0.0;
        
        // 准确率 = 采纳且有效的比例
        long accurate = feedbacks.stream()
            .filter(f -> f.getAccepted() && f.getEffectivenessScore() != null && f.getEffectivenessScore() > 0.7)
            .count();
        
        return (double) accurate / feedbacks.size();
    }

    /**
     * 检查是否需要触发调优
     */
    private boolean shouldTriggerTuning() {
        // 每积累 100 条反馈触发一次调优
        long totalFeedback = feedbackService.countToday();
        return totalFeedback % 100 == 0;
    }
```

```tsx
// components/intelligence/AiChatPanel.tsx

import { useState, useRef, useEffect } from 'react';
import { Input, Button, Avatar, Spin, message } from 'antd';
import { SendOutlined, AudioOutlined, PaperClipOutlined } from '@ant-design/icons';
import { useChat } from 'ai/react';

interface Message {
  id: string;
  role: 'user' | 'assistant' | 'system';
  content: string;
  toolCalls?: ToolCall[];
  documents?: Document[];
  timestamp: Date;
}

export function AiChatPanel() {
  const [input, setInput] = useState('');
  const messagesEndRef = useRef<HTMLDivElement>(null);
  
  const { messages, isLoading, append, reload, stop } = useChat({
    api: '/api/ai/chat',
    initialMessages: [
      {
        id: 'system-1',
        role: 'system',
        content: `你是服装供应链智能助手小云。你可以帮助用户：
          1. 查询订单、生产、库存、财务等业务数据
          2. 分析订单健康度、交付风险、成本偏差
          3. 执行催办、派单、采购、审批等操作
          4. 提供智能建议和最佳实践
          
          回答时请：
          - 简洁明了，重点突出
          - 提供具体数据和原因
          - 给出可执行的建议
          - 使用 Markdown 格式`,
        timestamp: new Date(),
      }
    ],
    onError: (error) => {
      message.error(`对话出错: ${error.message}`);
    },
  });

  // 自动滚动到底部
  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  }, [messages]);

  // 发送消息
  const handleSend = async () => {
    if (!input.trim() || isLoading) return;
    
    await append({
      id: Date.now().toString(),
      role: 'user',
      content: input.trim(),
      timestamp: new Date(),
    });
    
    setInput('');
  };

  // 快捷命令
  const quickCommands = [
    { label: '今日风险订单', command: '帮我查一下今天有哪些高风险订单' },
    { label: '待处理任务', command: '我有哪些待处理的任务？' },
    { label: '库存预警', command: '哪些物料库存不足？' },
    { label: '财务待审批', command: '帮我看看财务有哪些待审批的' },
  ];

  return (
    <div className="ai-chat-panel">
      {/* 消息列表 */}
      <div className="messages-container">
        {messages
          .filter(m => m.role !== 'system')
          .map((msg) => (
            <MessageBubble key={msg.id} message={msg} />
          ))}
        
        {isLoading && (
          <div className="loading-indicator">
            <Spin size="small" />
            <span>小云正在思考...</span>
          </div>
        )}
        
        <div ref={messagesEndRef} />
      </div>

      {/* 快捷命令 */}
      <div className="quick-commands">
        {quickCommands.map((cmd) => (
          <Button
            key={cmd.label}
            size="small"
            onClick={() => {
              setInput(cmd.command);
              handleSend();
            }}
          >
            {cmd.label}
          </Button>
        ))}
      </div>

      {/* 输入区域 */}
      <div className="input-area">
        <Input.TextArea
          value={input}
          onChange={(e) => setInput(e.target.value)}
          placeholder="问我任何问题..."
          autoSize={{ minRows: 1, maxRows: 4 }}
          onPressEnter={(e) => {
            if (!e.shiftKey) {
              e.preventDefault();
              handleSend();
            }
          }}
        />
        <Space>
          <Button icon={<AudioOutlined />} />
          <Button icon={<PaperClipOutlined />} />
          <Button 
            type="primary" 
            icon={<SendOutlined />}
            onClick={handleSend}
            loading={isLoading}
          >
            发送
          </Button>
        </Space>
      </div>
    </div>
  );
}

// 消息气泡
function MessageBubble({ message }: { message: Message }) {
  const isUser = message.role === 'user';
  
  return (
    <div className={`message-bubble ${isUser ? 'user' : 'assistant'}`}>
      {!isUser && (
        <Avatar className="avatar" style={{ background: '#1890ff' }}>
          云
        </Avatar>
      )}
      
      <div className="content">
        <div className="text">
          {isUser ? message.content : (
            <MarkdownRenderer content={message.content} />
          )}
        </div>
        
        {/* 工具调用结果 */}
        {message.toolCalls && message.toolCalls.length > 0 && (
          <div className="tool-results">
            {message.toolCalls.map((tool, index) => (
              <ToolResultCard key={index} tool={tool} />
            ))}
          </div>
        )}
        
        <div className="timestamp">
          {message.timestamp.toLocaleTimeString()}
        </div>
      </div>
    </div>
  );
}
```
