CREATE OR REPLACE FUNCTION search_similar_memories(
    p_tenant_id BIGINT,
    p_query_vector vector(1536),
    p_similarity_threshold FLOAT DEFAULT 0.8,
    p_limit INT DEFAULT 5,
    p_memory_type VARCHAR DEFAULT NULL
)
RETURNS TABLE (
    id BIGINT,
    memory_type VARCHAR(50),
    memory_title VARCHAR(255),
    memory_summary TEXT,
    memory_tags TEXT[],
    source_biz_type VARCHAR(50),
    source_biz_id BIGINT,
    quality_score DECIMAL(3,2),
    similarity FLOAT
) AS $$
BEGIN
    RETURN QUERY
    SELECT 
        m.id,
        m.memory_type,
        m.memory_title,
        m.memory_summary,
        m.memory_tags,
        m.source_biz_type,
        m.source_biz_id,
        m.quality_score,
        1 - (m.content_vector <=> p_query_vector) AS similarity
    FROM t_intelligence_memory m
    WHERE m.tenant_id = p_tenant_id
      AND (p_memory_type IS NULL OR m.memory_type = p_memory_type)
      AND m.content_vector IS NOT NULL
    ORDER BY m.content_vector <=> p_query_vector
    LIMIT p_limit;
END;
$$ LANGUAGE plpgsql;

-- ============================================
-- 2. 统计函数
-- ============================================

CREATE OR REPLACE FUNCTION get_intelligence_stats(
    p_tenant_id BIGINT,
    p_start_date DATE DEFAULT CURRENT_DATE - INTERVAL '7 days',
    p_end_date DATE DEFAULT CURRENT_DATE
)
RETURNS TABLE (
    stat_date DATE,
    signal_count BIGINT,
    action_count BIGINT,
    completed_action_count BIGINT,
    feedback_count BIGINT,
    avg_effectiveness DECIMAL(3,2)
) AS $$
BEGIN
    RETURN QUERY
    SELECT 
        DATE(s.signal_time) AS stat_date,
        COUNT(DISTINCT s.id) AS signal_count,
        COUNT(DISTINCT a.id) AS action_count,
        COUNT(DISTINCT CASE WHEN a.action_status = 'COMPLETED' THEN a.id END) AS completed_action_count,
        COUNT(DISTINCT f.id) AS feedback_count,
        AVG(f.effectiveness_score) AS avg_effectiveness
    FROM t_intelligence_signal s
    LEFT JOIN t_intelligence_action a ON a.signal_id = s.id
    LEFT JOIN t_intelligence_feedback f ON f.action_id = a.id
    WHERE s.tenant_id = p_tenant_id
      AND DATE(s.signal_time) BETWEEN p_start_date AND p_end_date
    GROUP BY DATE(s.signal_time)
    ORDER BY stat_date;
END;
$$ LANGUAGE plpgsql;

-- ============================================
-- 3. 建议效果统计函数
-- ============================================

CREATE OR REPLACE FUNCTION get_suggestion_effectiveness(
    p_tenant_id BIGINT,
    p_suggestion_type VARCHAR DEFAULT NULL
)
RETURNS TABLE (
    suggestion_type VARCHAR(50),
    total_suggestions BIGINT,
    total_feedbacks BIGINT,
    accepted_count BIGINT,
    adoption_rate DECIMAL(5,2),
    avg_effectiveness DECIMAL(3,2),
    avg_confidence DECIMAL(3,2)
) AS $$
BEGIN
    RETURN QUERY
    SELECT 
        s.suggestion_type,
        COUNT(DISTINCT s.id) AS total_suggestions,
        SUM(s.total_feedback) AS total_feedbacks,
        SUM(s.accepted_count) AS accepted_count,
        CASE 
            WHEN SUM(s.total_feedback) > 0 
            THEN ROUND(SUM(s.accepted_count)::DECIMAL / SUM(s.total_feedback) * 100, 2)
            ELSE 0 
        END AS adoption_rate,
        AVG(f.effectiveness_score) AS avg_effectiveness,
        AVG(s.confidence) AS avg_confidence
    FROM t_intelligence_suggestion s
    LEFT JOIN t_intelligence_feedback f ON f.suggestion_id = s.id
    WHERE s.tenant_id = p_tenant_id
      AND (p_suggestion_type IS NULL OR s.suggestion_type = p_suggestion_type)
    GROUP BY s.suggestion_type;
END;
$$ LANGUAGE plpgsql;

package com.fashion.supplychain.intelligence.service;

import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.http.*;
import org.springframework.stereotype.Service;
import org.springframework.web.client.RestTemplate;
import java.util.*;

@Slf4j
@Service
public class EmbeddingService {

    @Value("${ai.embedding.model:text-embedding-3-small}")
    private String embeddingModel;

    @Value("${ai.embedding.api-key:}")
    private String apiKey;

    @Value("${ai.embedding.base-url:https://api.openai.com/v1}")
    private String baseUrl;

    @Autowired
    private RestTemplate restTemplate;

    public float[] embed(String text) {
        if (text == null || text.isBlank()) {
            return new float[1536];
        }

        try {
            HttpHeaders headers = new HttpHeaders();
            headers.setContentType(MediaType.APPLICATION_JSON);
            headers.setBearerAuth(apiKey);

            Map<String, Object> body = new HashMap<>();
            body.put("model", embeddingModel);
            body.put("input", text);

            HttpEntity<Map<String, Object>> request = new HttpEntity<>(body, headers);
            
            ResponseEntity<Map> response = restTemplate.postForEntity(
                baseUrl + "/embeddings", 
                request, 
                Map.class
            );

            if (response.getStatusCode() == HttpStatus.OK && response.getBody() != null) {
                Map<String, Object> responseBody = response.getBody();
                List<Map<String, Object>> data = (List<Map<String, Object>>) responseBody.get("data");
                if (data != null && !data.isEmpty()) {
                    List<Double> embedding = (List<Double>) data.get(0).get("embedding");
                    float[] result = new float[embedding.size()];
                    for (int i = 0; i < embedding.size(); i++) {
                        result[i] = embedding.get(i).floatValue();
                    }
                    return result;
                }
            }
        } catch (Exception e) {
            log.error("Embedding failed: {}", e.getMessage());
        }

        return new float[1536];
    }

    public List<float[]> embedBatch(List<String> texts) {
        return texts.stream()
            .map(this::embed)
            .toList();
    }

    public String vectorToString(float[] vector) {
        StringBuilder sb = new StringBuilder("[");
        for (int i = 0; i < vector.length; i++) {
            if (i > 0) sb.append(",");
            sb.append(vector[i]);
        }
        sb.append("]");
        return sb.toString();
    }
}

            .signalPayload(request.getPayload())
            .signalTime(LocalDateTime.now())
            .sourceTerminal(request.getSourceTerminal())
            .processed(false)
            .createdBy(Long.parseLong(userId))
            .build();

        signal = signalService.save(signal);

        eventPublisher.publishEvent(new SignalCollectedEvent(signal));

        log.info("Signal collected: id={}, type={}, level={}", 
            signal.getId(), signal.getSignalType(), signal.getSignalLevel());

        return signal;
    }

    @Transactional
    public List<IntelligenceSignal> collectBatch(List<SignalRequest> requests) {
        return requests.stream()
            .map(this::collect)
            .toList();
    }

    private String calculateLevel(SignalRequest request) {
        return switch (request.getSignalType()) {
            case "ORDER_DELAY" -> {
                Integer delayDays = getDelayDays(request);
                yield delayDays != null && delayDays > 3 ? "CRITICAL" : "WARNING";
            }
            case "SCAN_ANOMALY" -> "ERROR";
            case "STOCK_ALERT" -> {
                Double shortageRate = getShortageRate(request);
                yield shortageRate != null && shortageRate > 0.5 ? "CRITICAL" : "WARNING";
            }
            case "FINANCE_DIFF" -> {
                Double diffAmount = getDiffAmount(request);
                yield diffAmount != null && Math.abs(diffAmount) > 10000 ? "CRITICAL" : "ERROR";
            }
            case "SYSTEM_ALERT" -> "CRITICAL";
            default -> "INFO";
        };
    }

    private Integer getDelayDays(SignalRequest request) {
        if (request.getPayload() instanceof Map) {
            Object days = ((Map<?, ?>) request.getPayload()).get("delayDays");
            if (days instanceof Number) {
                return ((Number) days).intValue();
            }
        }
        return null;
    }

    private Double getShortageRate(SignalRequest request) {
        if (request.getPayload() instanceof Map) {
            Object rate = ((Map<?, ?>) request.getPayload()).get("shortageRate");
            if (rate instanceof Number) {
                return ((Number) rate).doubleValue();
            }
        }
        return null;
    }

    private Double getDiffAmount(SignalRequest request) {
        if (request.getPayload() instanceof Map) {
            Object amount = ((Map<?, ?>) request.getPayload()).get("diffAmount");
            if (amount instanceof Number) {
                return ((Number) amount).doubleValue();
            }
        }
        return null;
    }

    @Data
    public static class SignalRequest {
        private String signalType;
        private String bizType;
        private Long bizId;
        private Object payload;
        private String sourceTerminal;
    }

    @Data
    @AllArgsConstructor
    public static class SignalCollectedEvent {
        private IntelligenceSignal signal;
    }
}

// IntelligenceConfigService.java
package com.fashion.supplychain.intelligence.service;

import com.baomidou.mybatisplus.extension.service.impl.ServiceImpl;
import com.fashion.supplychain.intelligence.entity.IntelligenceConfig;
import com.fashion.supplychain.intelligence.mapper.IntelligenceConfigMapper;
import org.springframework.cache.annotation.Cacheable;
import org.springframework.stereotype.Service;
import java.util.Map;
import java.util.HashMap;

@Service
public class IntelligenceConfigService extends ServiceImpl<IntelligenceConfigMapper, IntelligenceConfig> {

    @Cacheable(value = "intelligence-config", key = "#tenantId + ':' + #configKey")
    public boolean isEnabled(Long tenantId, String configKey) {
        IntelligenceConfig config = lambdaQuery()
            .eq(IntelligenceConfig::getTenantId, tenantId)
            .eq(IntelligenceConfig::getConfigKey, configKey)
            .one();
        
        if (config == null) {
            return getDefaultEnabled(configKey);
        }
        
        return config.getEnabled();
    }

    @Cacheable(value = "intelligence-config", key = "#tenantId + ':all'")
    public Map<String, Object> getAllConfigs(Long tenantId) {
        Map<String, Object> configs = new HashMap<>();
        
        list(lambdaQuery()
            .eq(IntelligenceConfig::getTenantId, tenantId)
            .eq(IntelligenceConfig::getEnabled, true))
            .forEach(config -> configs.put(config.getConfigKey(), config.getConfigValue()));
        
        return configs;
    }

    private boolean getDefaultEnabled(String configKey) {
        return switch (configKey) {
            case "smart.signal.enabled" -> true;
            case "smart.health-score.enabled" -> true;
            case "smart.action.auto-execute" -> false;
            case "smart.memory.vector-search" -> true;
            case "smart.learning.feedback" -> true;
            default -> false;
        };
    }
}
