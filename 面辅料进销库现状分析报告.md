```java
// StyleBomServiceImpl.java
@Override
public List<StyleBom> saveBomWithStockCheck(List<StyleBom> bomList, Integer productionQty) {
    for (StyleBom bom : bomList) {
        // 1. 计算需求量（含损耗）
        int requiredQty = calculateRequirement(bom, productionQty);
        
        // 2. 查询库存
        MaterialStock stock = materialStockService.findByMaterialAndColor(
            bom.getMaterialCode(), bom.getColor(), bom.getSize()
        );
        
        // 3. 计算状态
        int availableQty = stock != null ? (stock.getQuantity() - stock.getLockedQuantity()) : 0;
        if (availableQty >= requiredQty) {
            bom.setStockStatus("sufficient");  // 充足
        } else if (availableQty > 0) {
            bom.setStockStatus("insufficient"); // 不足
        } else {
            bom.setStockStatus("none");        // 无库存
        }
        
        bom.setAvailableStock(availableQty);
        bom.setRequiredPurchase(Math.max(0, requiredQty - availableQty));
    }
    this.saveBatch(bomList);
    return bomList;
}
```

#### Task 5: 库存预警系统 ⏱️ 2天

```java
// MaterialStockAlertService.java (新建)
@Service
public class MaterialStockAlertService {
    public List<Map<String, Object>> getLowStockAlerts() {
        return materialStockService.list(
            new LambdaQueryWrapper<MaterialStock>()
                .apply("(quantity - locked_quantity) < safety_stock")
        ).stream().map(stock -> {
            Map<String, Object> alert = new HashMap<>();
            alert.put("materialCode", stock.getMaterialCode());
            alert.put("availableQty", stock.getQuantity() - stock.getLockedQuantity());
            alert.put("safetyStock", stock.getSafetyStock());
            alert.put("shortfall", stock.getSafetyStock() - (stock.getQuantity() - stock.getLockedQuantity()));
            return alert;
        }).collect(Collectors.toList());
    }
}
```
