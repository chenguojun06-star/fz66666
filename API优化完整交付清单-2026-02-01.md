#### 6. 迁移统计功能

```typescript
// 在适配器中添加统计逻辑
const deprecatedStats = new Map<string, number>();

function trackDeprecatedUsage(endpoint: string) {
  const count = deprecatedStats.get(endpoint) || 0;
  deprecatedStats.set(endpoint, count + 1);
}

export function getDeprecationStats() {
  return Array.from(deprecatedStats.entries())
    .map(([endpoint, count]) => ({ endpoint, count }))
    .sort((a, b) => b.count - a.count);
}
```

#### 7. 迁移进度报告生成器

```typescript
// scripts/generate-migration-report.ts
import { getDeprecationStats } from '@/utils/api';

function generateReport() {
  const stats = getDeprecationStats();
  const total = stats.reduce((sum, item) => sum + item.count, 0);
  
  console.log(`废弃端点调用统计（过去7天）`);
  console.log(`总调用次数：${total}`);
  console.log(`\nTop 10 高频端点：`);
  
  stats.slice(0, 10).forEach(({ endpoint, count }) => {
    const percentage = ((count / total) * 100).toFixed(2);
    console.log(`  ${endpoint}: ${count}次 (${percentage}%)`);
  });
}
```
