## 🔧 标准迁移模板

### 简单按钮（1-2 个操作）

**修改前**：
```tsx
{
  title: '操作',
  key: 'action',
  render: (_: any, record: any) => (
    <Space>
      <Button onClick={() => handleEdit(record)}>编辑</Button>
      <Button danger onClick={() => handleDelete(record)}>删除</Button>
    </Space>
  )
}
```

**修改后**：
```tsx
{
  title: '操作',
  key: 'action',
  width: 150,
  fixed: 'right' as const,
  render: (_: any, record: any) => (
    <RowActions
      actions={[
        { key: 'edit', label: '编辑', onClick: () => handleEdit(record) },
        { key: 'delete', label: '删除', danger: true, onClick: () => handleDelete(record) },
      ]}
    />
  )
}
```

### 条件渲染（3-5 个操作）

**修改前**：
```tsx
{
  title: '操作',
  key: 'action',
  render: (_: any, record: any) => {
    const actions = [];
    if (record.status === 'draft') {
      actions.push(<Button onClick={() => handleEdit(record)}>编辑</Button>);
    }
    if (canDelete) {
      actions.push(<Button danger onClick={() => handleDelete(record)}>删除</Button>);
    }
    return <Space>{actions}</Space>;
  }
}
```

**修改后**：
```tsx
{
  title: '操作',
  key: 'action',
  width: 180,
  fixed: 'right' as const,
  render: (_: any, record: any) => {
    const actions = [];
    if (record.status === 'draft') {
      actions.push({ key: 'edit', label: '编辑', onClick: () => handleEdit(record) });
    }
    if (canDelete) {
      actions.push({ key: 'delete', label: '删除', danger: true, onClick: () => handleDelete(record) });
    }
    return <RowActions actions={actions} />;
  }
}
```
